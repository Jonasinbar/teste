  <script>
    function handleRowClick(rowData) {
      alert(`Vous avez cliquÃ© sur : ${rowData["Nom"]}`);
    }

    fetch('/api/data')
      .then(res => res.json())
      .then(data => {
        if (data.length === 0) return;

        const orderedHeaders = ["Nom", "Age", "Ville"]; // ðŸ‘ˆ Change column order here

        const headerRow = document.getElementById('table-headers');
        orderedHeaders.forEach(h => {
          const th = document.createElement('th');
          th.textContent = h;
          th.className = "px-6 py-3 text-left text-sm font-semibold text-gray-700";
          headerRow.appendChild(th);
        });

        const body = document.getElementById('table-body');
        data.forEach(row => {
          const tr = document.createElement('tr');
          tr.className = "hover:bg-gray-100 cursor-pointer transition-colors";
          tr.onclick = () => handleRowClick(row);

          orderedHeaders.forEach(h => {
            const td = document.createElement('td');
            td.textContent = row[h];
            td.className = "px-6 py-4 text-sm text-gray-800 whitespace-nowrap";
            tr.appendChild(td);
          });

          body.appendChild(tr);
        });
      });
  </script>
